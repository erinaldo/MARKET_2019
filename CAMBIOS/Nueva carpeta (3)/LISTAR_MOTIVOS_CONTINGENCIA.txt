CREATE PROCEDURE [dbo].[LISTAR_MOTIVOS_CONTINGENCIA]
	@SW INT,@CADENA VARCHAR(250),@CAMPO VARCHAR(250)
  AS
BEGIN
	DECLARE @CAD VARCHAR(2500)
	
	SET @CAD="SELECT COD_MCONTINGENCIA AS Codigo, DESC_MCONTINGENCIA as Descripcion "+
		" FROM MOTIVO_CONTINGENCIA WHERE COD_MCONTINGENCIA<>0 "
	--TODOS
	PRINT @SW
	--IF @SW<>2  SET @CAD=@CAD+"AND STAT_RUBRO="+CONVERT(CHAR(1),@SW)+""

	--CAMPO A BUSCAR
	IF @CADENA<>'' SET @CAD=@CAD+" AND "+@CAMPO+" LIKE '%"+@CADENA+"%'" 

	
	SET @CAD=@CAD+" ORDER BY DESC_MCONTINGENCIA ASC"

	PRINT @CAD
	EXEC(@CAD)
END







GO
