CREATE PROCEDURE [dbo].[IAM_ESTADO_FELECTRONICA_CONTINGENCIA] 
	
	@ID_VENTASM BIGINT,
	@TIPO INT,
	@SW INT OUTPUT
	
AS

BEGIN TRAN IAM_ACTUALIZAR
SET @SW=0


	BEGIN
		UPDATE VENTAS_MANUAL SET FELECTRONICA=@TIPO WHERE ID_VENTASM=@ID_VENTASM
	END

IF @@ERROR!=0
	BEGIN
		ROLLBACK TRAN IAM_ACTUALIZAR
		SET @SW=1
		RETURN
	END				

COMMIT TRAN IAM_ACTUALIZAR
RETURN 




GO
