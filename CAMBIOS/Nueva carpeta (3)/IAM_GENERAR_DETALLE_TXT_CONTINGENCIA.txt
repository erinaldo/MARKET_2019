CREATE PROCEDURE [dbo].[IAM_GENERAR_DETALLE_TXT_CONTINGENCIA]    
@ID_VENTASM BIGINT
  AS        
BEGIN        
SELECT     VENTASD_MANUAL.ID_VENTASM AS ITEM, VENTASD_MANUAL.COD_ART AS AMO2ART, VENTASD_MANUAL.CANT_VENTASDM AS AMO2CANTI, ARTICULOS.DESC_ART, CONVERT(NUMERIC(18, 2), 
                      (VENTASD_MANUAL.CANT_VENTASDM * VENTASD_MANUAL.PU_VENTASDM) / (1 + VENTAS_MANUAL.PIGV / 100)) AS AMO2STOT, CONVERT(NUMERIC(18, 2), 
                      VENTASD_MANUAL.PU_VENTASDM / (1 + VENTAS_MANUAL.PIGV / 100)) AS AMO2PREC, 'NIU' AS AUMESUNAT, VENTASD_MANUAL.PU_VENTASDM AS AVTAPREC, ISNULL(ARTICULOS.COD_SUNAT, 
                      '') AS COD_SUNAT
FROM         VENTASD_MANUAL INNER JOIN
                      ARTICULOS ON VENTASD_MANUAL.COD_ART = ARTICULOS.COD_ART INNER JOIN
                      VENTAS_MANUAL ON VENTASD_MANUAL.ID_VENTASM = VENTAS_MANUAL.ID_VENTASM INNER JOIN
                      MEDIDA ON ARTICULOS.COD_MEDIDA = MEDIDA.COD_MEDIDA
WHERE     (VENTASD_MANUAL.ID_VENTASM = @ID_VENTASM)



END
