CREATE PROCEDURE  [dbo].[LISTAR_TMP_DETALLE_MOV]
@CORRNBR BIGINT
AS

DECLARE @sql AS varchar(8000)


SET @SQL="SELECT     TMP_DETALLE.ITEM AS Id, ARTICULOS.COD_ART AS Cod, ARTICULOS.DESC_ART AS Producto, MEDIDA.DESC_MEDIDA AS Und, TMP_DETALLE.ATMPCANT AS Cant, 
                      TMP_DETALLE.ATMPPREC AS Pu, TMP_DETALLE.ATMPCANT * TMP_DETALLE.ATMPPREC AS Total
FROM         TMP_DETALLE INNER JOIN
                      ARTICULOS ON TMP_DETALLE.COD_ART = ARTICULOS.COD_ART INNER JOIN
                      MEDIDA ON ARTICULOS.COD_MEDIDA = MEDIDA.COD_MEDIDA "+
	" WHERE CORRNBR="+CONVERT(VARCHAR,@CORRNBR)+""+
	" ORDER BY ITEM"

PRINT @SQL
--EXEC sp_executesql @sql
exec(@sql)



GO
