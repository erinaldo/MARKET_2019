CREATE PROCEDURE [dbo].[IAM_ESTADO_FELECTRONICA] 
	
	@COD_DOC CHAR(3),
	@NRODOCU NUMERIC(10,0),
	@TIPO INT,
	@SW INT OUTPUT
	
AS

BEGIN TRAN IAM_ACTUALIZAR
SET @SW=0
UPDATE VENTACAB SET FELECTRONICA=@TIPO WHERE COD_DOC=@COD_DOC AND NRODOCU=@NRODOCU

IF @@ERROR!=0
	BEGIN
		ROLLBACK TRAN IAM_ACTUALIZAR
		SET @SW=1
		RETURN
	END				

COMMIT TRAN IAM_ACTUALIZAR
RETURN
