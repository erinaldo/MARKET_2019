CREATE PROCEDURE [dbo].[BUSCAR_CAJA]
@ID_CAJA INT
AS

SELECT     TIPOMOVI, FECHA_CAJA, MON_CAJA, MONTO_CAJA, OBSERV_CAJA, TRANSF_CAJA
FROM         MOVICAJA
WHERE     (ID_CAJA =@ID_CAJA)
