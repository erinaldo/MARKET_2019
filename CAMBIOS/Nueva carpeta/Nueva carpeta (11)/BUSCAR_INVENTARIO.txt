CREATE PROCEDURE [dbo].[BUSCAR_INVENTARIO]
	@COD_INV BIGINT,@COD_USER VARCHAR(10)
  AS
BEGIN
		
	SELECT COD_INV, FECHA_INV, COD_ALMACEN, OBSERV_INV, STAT_INV
	FROM INVENTARIO
	WHERE (COD_INV = @COD_INV)
	
	--LLENAMOS EL TEMP
	DELETE FROM TEMP_INVENTARIO WHERE COD_USER=@COD_USER
	INSERT INTO TEMP_INVENTARIO(COD_USER,ITEM,COD_ART,COSTO_ART,CANT_ART,STOCK_ART)
	SELECT @COD_USER,ITEM_INVD, COD_ART, COSTO_ART, CANT_INVD, STOCK_INVD
	FROM INVENTARIOD
	WHERE (COD_INV = @COD_INV)
END
GO
