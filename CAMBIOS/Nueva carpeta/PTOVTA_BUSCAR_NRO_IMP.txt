CREATE   PROCEDURE [dbo].[PTOVTA_BUSCAR_NRO_IMP]
(
	@COD_DOC VARCHAR(3),
	@TERMINAL VARCHAR(20),
	@PORT INT OUTPUT

)
AS
BEGIN 
SET NOCOUNT ON;
 SELECT  @PORT= dbo.PTOVTAD.NRO_PAPEL
	FROM         dbo.PTOVTA INNER JOIN
                      dbo.PTOVTAD ON dbo.PTOVTA.APTOCODI = dbo.PTOVTAD.APTOCODI
	WHERE     (dbo.PTOVTA.APTOSTAT = 0) AND COD_DOC=@COD_DOC AND  (PTOVTA.APTOTERM = @TERMINAL)
END

SET @PORT=ISNULL(@PORT,0)
PRINT @PORT
RETURN
GO
