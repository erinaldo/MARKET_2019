ALTER PROCEDURE [dbo].[CONSULTAR_COMITE_CANT]
	@FI CHAR(8),@FF CHAR(8)
  AS
BEGIN
	SELECT A.FECPROCESO AS 'Fecha', DAY(A.FECPROCESO) AS 'Dia', 
SUM(CAST( B.CANTIDAD AS DECIMAL(18, 2))) AS 'Cant Ventas', 
SUM(CAST (B.CANTIDAD AS DECIMAL(18, 2)))
                      AS 'Cant Ventas LT'
FROM         VENTACAB A INNER JOIN
                      VENTADET B ON A.COD_DOC = B.COD_DOC AND A.NRODOCU = B.NRODOCU
    WHERE     (A.FECPROCESO BETWEEN @FI AND @FF) AND (A.STAT_VENTA = 0)
      GROUP BY A.FECPROCESO
ORDER BY 'FECHA'
END
