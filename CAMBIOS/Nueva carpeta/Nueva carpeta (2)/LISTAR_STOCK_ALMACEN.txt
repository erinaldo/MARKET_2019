CREATE PROCEDURE [dbo].[LISTAR_STOCK_ALMACEN]
	@COD_ART CHAR(15)
	
  AS
BEGIN
	SELECT ALMACENES.DESC_ALMACEN AS Almacen,DBO.STOCK_TOTAL_ALMACEN(@COD_ART,ALMACENES.COD_ALMACEN) AS Stock
FROM         ALMACENES 
WHERE      (ALMACENES.STAT_ALMACEN = 0)
		
END
