CREATE PROCEDURE [dbo].[IMP_X_RESUMEN_STOCK]
@PTOVTA VARCHAR(30)
AS

DECLARE @COD_ALMACEN BIGINT
--AVERIGUAMOS EL ALMACEN
SELECT @COD_ALMACEN=COD_ALMACEN FROM PTOVTA WHERE APTOTERM=@PTOVTA AND APTOSTAT=0
SET @COD_ALMACEN=ISNULL(@COD_ALMACEN,0)

SELECT     COD_ART AS Codigo, DESC_ART as Descripcion,DBO.STOCK_TOTAL_ALMACEN(COD_ART,@COD_ALMACEN) AS STOCK
FROM         ARTICULOS
WHERE     (STAT_ART = 0)
