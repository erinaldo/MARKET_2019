CREATE PROCEDURE [dbo].[BUSCAR_TRANSF_TMOV]
@COD_TMOV CHAR(2),@TRANSF BIT OUTPUT
 AS

SET @TRANSF=0

IF (SELECT COUNT(COD_TMOV) FROM TIPOMOVI WHERE COD_TMOV=@COD_TMOV AND TRANSF_TMOV=1)>0
	BEGIN
		SET @TRANSF=1
	END
