ALTER PROCEDURE [dbo].[IAM_RPT_GASTOS] 
@FI CHAR(8),@FF CHAR(8)

AS

SELECT     CAJA_OFICINA.COD_MCAJA, PLAN_CUENTASD.DESC_PLANCTAD, CAJA_OFICINA.MONTO_MCAJA, CAJA_OFICINA.OBSERV_MCAJA, CAJA_OFICINA.MON_MCAJA, 
                      CAJA_OFICINA.FECHA_MCAJA
FROM         CAJA_OFICINA INNER JOIN
                      PLAN_CUENTASD ON CAJA_OFICINA.COD_PLANCTA = PLAN_CUENTASD.COD_PLANCTA AND CAJA_OFICINA.SUB_PLANCTA = PLAN_CUENTASD.SUB_PLANCTA
WHERE     (CAJA_OFICINA.TIPO_MCAJA = 'S') AND (CAJA_OFICINA.STAT_MCAJA = 0)
		AND  (FECHA_MCAJA BETWEEN @FI AND @FF)
