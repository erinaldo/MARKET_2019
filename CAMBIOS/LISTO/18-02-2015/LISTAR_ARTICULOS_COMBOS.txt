CREATE PROCEDURE [dbo].[LISTAR_ARTICULOS_COMBOS]
@COD VARCHAR(15)
 AS
SELECT     Combos.COD_ARTC AS Codigo, ARTICULOS.DESC_ART AS Descripcion, MEDIDA.DESC_MEDIDA AS Medida, Combos.CANT_COMBO AS Cantidad
FROM         Combos INNER JOIN
                      ARTICULOS ON Combos.COD_ARTC = ARTICULOS.COD_ART LEFT OUTER JOIN
                      MEDIDA ON ARTICULOS.COD_MEDIDA = MEDIDA.COD_MEDIDA
WHERE     (Combos.COD_ART = @COD)

GO
