CREATE PROCEDURE [dbo].[BUSCAR_PERMISO]
@NIVEL1 VARCHAR(6),@COD_USER VARCHAR(10),@SW BIT OUTPUT
 AS

SET @SW=0

IF (SELECT COUNT(COD_USER) FROM PERMISOS WHERE COD_USER=@COD_USER AND NIVEL1=@NIVEL1)>0
	BEGIN
		SET @SW=1
	END


RETURN

GO
